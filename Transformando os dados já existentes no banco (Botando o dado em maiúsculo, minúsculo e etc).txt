            // Carrega os "Customers" que tiverem o id MAIOR que 1 e depois ADICIONA "Dear " atrás do nome do 	    objeto e coloca o nome já existente em MAIÚSCULO, depois coloca o "adress"(endereço) em MAIÚSCULO 	    também
            $list = Customer::where('id', '>', 1)->transform( function($object) {
                $object->name    = 'Dear '. strtoupper($object->name);
                $object->address = strtoupper($object->address);
            });
                                 
            foreach ($list as $customer)
            {
                echo $customer->id . ' - ' . $customer->name . ' - ' . $customer->address. '<br>';
            }
            TTransaction::close(); // fecha a transação.
